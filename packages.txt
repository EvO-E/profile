# - http://www.trinitydesktop.org/installation.php#ubuntu
# - sudo add-apt-repository ppa:nilarimogard/webupd8 (for up-to-date Audacious)
# - sudo add-apt-repository ppa:ubuntu-mozilla-security/ppa (for up-to-date Firefox)
# - sudo add-apt-repository ppa:ubuntu-wine/ppa (for up-to-date Wine)
#
